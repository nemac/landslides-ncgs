# Python dependencies for the DEFNS Alerts data pipeline (scripts/refresh.py).
#
# Install once into your environment:
#     conda activate EWS
#     pip install -r scripts/requirements.txt
#
# Or for a fresh conda env (recommended for GitHub Actions in Phase 3b):
#     conda env create -n defns-refresh \
#         -c conda-forge python=3.11 geopandas rasterio libgdal-grib requests
#
# rasterio + libgdal-grib are needed to read MRMS GRIB2 files. libgdal-grib
# is a separate conda-forge plugin in GDAL 3.9+ (was bundled in earlier
# versions). If you're on Windows and getting "gdal_GRIB.dll not available"
# errors, that's the missing plugin; install via:
#     conda install -c conda-forge libgdal-grib
